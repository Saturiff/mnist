int NextInt32(FILE* filePtr)
{
	byte buffer[sizeof(int)] = { 0 };
	fread(buffer, sizeof(byte), sizeof(int), filePtr);
	int num = 0;
	for (int i = 0; i < sizeof(int); i++) num |= (int)buffer[i] << (sizeof(int) - i - 1) * 8; // bit offset 0, 8, 16, 24
	return num;
}
